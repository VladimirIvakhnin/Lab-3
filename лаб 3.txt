    // Завдання 1: визначити кількість днів у році
void task_if128()
{
    int year;
    cout << "Введіть рік: ";
    cin >> year;

    if (cin.fail()) {
        cout << "Невірне введення!" << endl;
        return ;
    }

    bool leap = (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);

    if (leap)
        cout << "Цього року 366 днів (високосний рік)." << endl;
    else
        cout << "Цей рік має 365 днів (звичайний рік)." << endl;
}

// Завдання 2: варіант 13 – перевірити належність точки області
void task_geom13() {
    double x, y, r;
    cout << "Введіть координати точки (x, y): ";
    cin >> x >> y;
    cout << "Введіть радіус r: ";
    cin >> r;

    // Перевірка правильності введення
    if (cin.fail() || r <= 0) {
        cerr << "Некоректні дані!" << endl;
        return;
    }

    // Обчислення квадрату відстані до центра кола
    double circle = x * x + y * y;

    // Умова належності точки області:
    // точка всередині кола (x^2 + y^2 <= r^2)
    // і при цьому в 1-й або 3-й чверті (тобто x*y >= 0)
    bool inArea = (circle <= r * r) && (x * y >= 0);

    // Виведення результату
    if (inArea)
        cout << "Точка належить до області (варіант 13, зелена частина)." << endl;
    else
        cout << "Точка НЕ належить до області (варіант 13)." << endl;
}
